```jsx
import * as React from 'react';
import { ThemedImage, Paragraph, ThemedPage } from 'react-pdf-builder';
import { PDFViewer, Document } from '@react-pdf/renderer';

export function Example() {
  return (
    <PDFViewer style={{ width: 420, height: 600 }}>
      <Document>
        <ThemedPage size="LETTER">
          <ThemedImage
            className="border p-3 rounded mb-3"
            src="https://justinmahar.github.io/react-pdf-builder/images/scootie-460.png"
            style={{ width: 230 }}
          />
          <Paragraph>
            This image uses the "border p-3 rounded mb-3" class names from the theme to achieve a thumbnail style with a
            bottom margin.
          </Paragraph>
        </ThemedPage>
      </Document>
    </PDFViewer>
  );
}
```
