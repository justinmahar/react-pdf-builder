## ThemedText Example

<DocsPDF>
  <ThemedText className="p-3 border">This is themed text with padding and a border from the theme.</ThemedText>
</DocsPDF>

```jsx
import * as React from 'react';
import { ThemedText, ThemedPage } from 'react-pdf-builder';
import { PDFViewer, Document } from '@react-pdf/renderer';

export function Example() {
  return (
    <PDFViewer style={{ width: 420, height: 600 }}>
      <Document>
        <ThemedPage size="LETTER">
          <ThemedText className="p-3 border">This is themed text with padding and a border from the theme.</ThemedText>
        </ThemedPage>
      </Document>
    </PDFViewer>
  );
}
```
